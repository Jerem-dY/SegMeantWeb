        s = ""
        s += json.dumps(self.txt, indent=indent, ensure_ascii=False)
        s += json.dumps(self.tags, indent=indent, ensure_ascii=False)

        #sert à récupérer le noeud le plus bas d'un certains type
        r = self

        if len(self.children)>0:
            if isinstance(self.children[0], childType):
                r = self.children[0]
            else:
                for child in self.children:
                    r = child.decapsulateToType(childType)

        else: 
            return None

        return r
